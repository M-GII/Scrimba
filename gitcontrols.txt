# ----------------------------------------------
# GitHub + VSCode Multi-Device Workflow (Bash Style)
# ----------------------------------------------

# 1. Git Setup (run this once on any new computer)
git config --global user.name "Your Name"
git config --global user.email "your@email.com"

# 2. Clone Your Repo from GitHub (M-GII/Scrimba--Projects)
# Use this ONCE on any new computer

# Option 1: Using HTTPS
git clone https://github.com/M-GII/Scrimba--Projects.git

# Option 2: Using SSH (only if you’ve set up SSH keys)
# git clone git@github.com:M-GII/Scrimba--Projects.git

# Navigate into the project folder and open in VSCode
cd Scrimba--Projects
code .

# 3. Check Remote Repo (to confirm it points to GitHub)
git remote -v
# Should show:
# origin  https://github.com/M-GII/Scrimba--Projects.git (fetch)
# origin  https://github.com/M-GII/Scrimba--Projects.git (push)

# 4. Make Changes Locally and Push to GitHub
git add .
git commit -m "Your commit message"
git push origin main

# 5. Pull Latest Changes on Another Computer
git pull origin main

# 6. Open VSCode Terminal
# Press this shortcut inside VSCode:
# Ctrl + ` (backtick, usually under the Esc key)
# OR click 'Terminal' > 'New Terminal' from the top menu

# 7. SSH Setup (optional: only if you want to avoid typing GitHub token every time)
ssh-keygen -t ed25519 -C "your@email.com"
cat ~/.ssh/id_ed25519.pub
# Copy the output and paste it at:
# https://github.com/settings/keys → "New SSH Key"
